(87 variables total) // Step 5 — Critical routing key active_jurisdictions: string[] // drives ALL clause activation } export interface BuiltPrompt { system: string // full system prompt for Claude user: string // user turn content metadata: { policy_type: PolicyType jurisdictions: Jurisdiction[] clauses_activated: string[] // audit trail estimated_tokens: number version: string } }
// ============================================================ // PolicyPen — Jurisdiction Router // Questionnaire answers → applicable laws (Set-based dedup) // ============================================================ export function resolveJurisdictions(q: Questionnaire): Jurisdiction[] { const laws = new Set<Jurisdiction>() const regions = q.active_jurisdictions.map(r => r.toLowerCase()) // ── Geographic mapping ────────────────────────────────────── if (regions.some(r => r.includes('eu') || r.includes('european'))) { laws.add('GDPR') laws.add('ePrivacy') } if (regions.some(r => r.includes('california'))) { laws.add('CCPA') laws.add('CPRA') if (q.auto_renewal === 'yes') laws.add('CA_ARL') } if (regions.some(r => r.includes('canada'))) { laws.add('PIPEDA') laws.add('CASL') } // ... UK_GDPR, LGPD, PDPA, POPIA, AU_PRIVACY if (regions.some(r => r.includes('global'))) { // Conservative baseline: add all major laws ['GDPR', 'CCPA', 'PIPEDA', 'ePrivacy'].forEach(j => laws.add(j as Jurisdiction)) } // ── Feature-triggered laws ──────────────────────────────── if (q.minimum_age === 'all_ages') laws.add('COPPA') if (q.regulated_industry.includes('healthcare')) laws.add('HIPAA') if (q.ugc_type !== 'none') laws.add('DMCA') if (q.marketing_channels.length > 0) laws.add('CAN_SPAM') if (q.marketing_channels.includes('sms')) laws.add('TCPA') if (q.business_model.some(m => m !== 'free')) { laws.add('FTC') if (laws.has('GDPR')) laws.add('EU_CRD') } return Array.from(laws) }
// clause-library.ts — showing 2 of 28 clauses as example // Each clause: id, jurisdictions[], condition(), instruction, priority export const PRIVACY_CLAUSES: Clause[] = [ // GDPR Art.6 Legal Basis — mandatory for any EU/UK processing { id: 'gdpr_legal_basis', jurisdiction: ['GDPR', 'UK_GDPR'], condition: (q) => q.legal_basis.length > 0, priority: 'mandatory', instruction: `Include a "Legal Basis for Processing" section under GDPR Article 6. For each legal basis in {{legal_basis}}: - consent: account data, marketing, non-essential cookies - contract: data required to deliver the service - legal_obligation: financial records, fraud prevention - legitimate_interests: security, analytics, abuse prevention State that where consent is the basis, users may withdraw at any time without affecting prior lawful processing.` }, // GDPR Art.9 Special Category — high risk, explicit consent req { id: 'gdpr_special_category', jurisdiction: ['GDPR', 'UK_GDPR'], condition: (q) => q.special_category_data.length > 0 && !q.special_category_data.includes('none'), priority: 'mandatory', instruction: `Include a "Special Category Data" section (GDPR Art.9). List each sensitive data type: {{special_category_data}}. State explicit consent is required (Art.9(2)(a)). Users can withdraw consent for special category data. Describe heightened security measures applied.` }, // ... 26 more clauses across all 4 policy types ] export function getClausesForPolicy( policyType: PolicyType, jurisdictions: Jurisdiction[], q: Questionnaire ): Clause[] { const library = { privacy: PRIVACY_CLAUSES, tos: TOS_CLAUSES, cookie: COOKIE_CLAUSES, refund: REFUND_CLAUSES }[policyType] return library.filter(clause => { const hasJurisdiction = clause.jurisdiction.some(j => jurisdictions.includes(j)) const meetsCondition = clause.condition(q) return hasJurisdiction && meetsCondition }) }
// builder.ts — The orchestration core // Variable interpolation: {{company_name}} → "Acme Inc." // Also translates enums: '12_months' → 'fees paid in last twelve months' function interpolate(template: string, q: Questionnaire): string { const vars: Record<string, string> = { company_name: q.company_name, product_name: q.product_name, liability_cap: ({ '12_months': 'fees paid in the twelve (12) months preceding the claim', '3_months': 'fees paid in the three (3) months preceding the claim', '100_usd': 'one hundred US dollars ($100)', 'none': 'the maximum extent permitted by applicable law', })[q.liability_cap], // ... all 87 variables with enum translation } return template.replace(/\{\{(\w+)\}\}/g, (_, key) => vars[key] ?? `[${key}]`) } // The main export — called from generate.ts and the API route export function buildPrompt( policyType: PolicyType, questionnaire: Questionnaire ): BuiltPrompt { // Step 1: Route jurisdictions const jurisdictions = resolveJurisdictions(questionnaire) // Step 2: Get applicable clauses const clauses = getClausesForPolicy(policyType, jurisdictions, questionnaire) // Step 3: Assemble system prompt const system = [ buildBaseSystem(questionnaire, policyType, jurisdictions), policySpecificSystem(policyType, questionnaire), buildClauseInstructions(clauses, questionnaire), ].join('\n') return { system, user: buildUserTurn(policyType, questionnaire, jurisdictions), metadata: { policy_type: policyType, jurisdictions, clauses_activated: clauses.map(c => c.id), estimated_tokens: Math.ceil((system.length) / 4), version: '1.0.0', }, } }
// generate.ts — Claude API integration with cost tracking const PRICING = { input_per_1k: 0.003, // claude-sonnet-4 rates output_per_1k: 0.015, } export async function generatePolicy( policyType: PolicyType, questionnaire: Questionnaire, onStream?: (chunk: string) => void ): Promise<GenerationResult> { const start = Date.now() const { system, user, metadata } = buildPrompt(policyType, questionnaire) let fullContent = '' const stream = await client.messages.stream({ model: 'claude-sonnet-4-20250514', max_tokens: 4096, system, messages: [{ role: 'user', content: user }], }) for await (const chunk of stream) { if (chunk.type === 'content_block_delta') { fullContent += chunk.delta.text onStream?.(chunk.delta.text) // → SSE to client } } const { input_tokens, output_tokens } = (await stream.finalMessage()).usage const cost = (input_tokens / 1000 * PRICING.input_per_1k) + (output_tokens / 1000 * PRICING.output_per_1k) return { html: fullContent, policy_type: policyType, tokens_input: input_tokens, tokens_output: output_tokens, cost_usd: Math.round(cost * 10000) / 10000, jurisdictions: metadata.jurisdictions, clauses_activated: metadata.clauses_activated, duration_ms: Date.now() - start, } }
// app/api/generate/route.ts — Next.js SSE streaming endpoint export const runtime = 'nodejs' // streaming requires nodejs, not edge export const maxDuration = 60 // 60s hard timeout export async function POST(req: Request) { // 1. Auth — Clerk validates JWT server-side const { userId } = auth() if (!userId) return Response.json({ error: 'Unauthorized' }, { status: 401 }) // 2. Cache check — return if < 24h old, no law updates const existing = await getCachedPolicy(productId, policyType) if (existing && !hasLawUpdate(existing)) { return Response.json({ cached: true, html: existing.content_html }) } // 3. Stream from Claude — ReadableStream → SSE const encoder = new TextEncoder() let accumulatedHtml = '' const stream = new ReadableStream({ async start(controller) { const result = await generatePolicy(policyType, questionnaire, (chunk) => { accumulatedHtml += chunk controller.enqueue( encoder.encode(`data: ${JSON.stringify({ chunk })}\n\n`) ) } ) // 4. Save to Supabase after completion await supabase.from('policies').insert({ product_id: productId, policy_type: policyType, content_html: accumulatedHtml, version: (existing?.version ?? 0) + 1, tokens_used: result.tokens_input + result.tokens_output, }) controller.enqueue(encoder.encode( `data: ${JSON.stringify({ done: true, cost_usd: result.cost_usd })}\n\n` )) controller.close() } }) return new Response(stream, { headers: { 'Content-Type': 'text/event-stream' } }) }

Legitimate interests → security, analytics, abuse prevention. State that consent may be withdrawn at any time without affecting prior lawful processing.
ccpa_rights mandatory
CCPACPRA
condition: always (when jurisdiction includes CCPA)
Include "California Privacy Rights" covering: Right to Know (categories, sources, purposes), Right to Delete (45-day response SLA), Right to Correct (CPRA), Right to Opt-Out of Sale/Sharing with "Do Not Sell or Share My Personal Information" link, Right to Non-Discrimination. If {{data_selling}} is 'no': explicitly state "We do not sell or share your personal information." Provide contact at {{contact_email}}.
tos_arbitration mandatory
FTC
condition: q.dispute_resolution === 'arbitration' || 'arbitration_opt_out'
Binding arbitration clause for non-EU/UK users only. CRITICAL: EU/UK consumers cannot be forced into arbitration — explicitly exempt them and preserve their court rights. US users: AAA Commercial Rules, 30-day informal resolution first, waiver of class action, opt-out within 30 days by emailing {{contact_email}}.
refund_eu_cooling_off mandatory
EU_CRDUK_GDPR
condition: q.active_jurisdictions includes 'eu' or 'uk'
EU/UK Right of Withdrawal: consumers have 14 days to withdraw under EU CRD Art.9. For digital services where user requests immediate access: waiver of withdrawal right — must be confirmed explicitly at checkout. If {{refund_policy}} is 'no_refunds': EU/UK consumers are exempt and retain statutory rights unless validly waived. Refund processed within 14 days of valid withdrawal.
ai_processing mandatory
GDPR Art.22CCPA
condition: q.ai_features.some(f => f !== 'none') || q.third_parties includes AI providers
AI Processing disclosure: if third_parties includes Anthropic/OpenAI, disclose that product features are AI-powered and user inputs may be processed by AI sub-processors with appropriate safeguards. If ai_features includes automated_decisions: include GDPR Art.22 right to not be subject to solely automated decisions, right to human review, and explanation of decision logic.
